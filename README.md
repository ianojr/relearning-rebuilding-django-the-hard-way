# Relearning Django and Django rest Framework the hard way for systems engineering.


# 1.Install pipenv - To manage venv and projects dependencies simultaneously.
pip install pipenv

# 2. Install django using pipenv
pipenv install django
- Force Pipenv to ignore the active environment - "set PIPENV_IGNORE_VIRTUALENVS=1"


# 3. Run the venv
pipenv shell



## 4. Create a django Project (Model(Data layer), View(Business Layer) Template(Presentation Layer) Architecture)
django-admin startproject firstproject
